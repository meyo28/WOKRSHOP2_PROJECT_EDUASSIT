def factorial_iterative(n):
    """
    Calculate factorial using iterative approach.
    Args:
        n: Non-negative integer
    Returns:
        Factorial of n
    """
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers")
    if n == 0:
        return 1
    
    result = 1
    for i in range(2, n + 1):
        result *= i
    return result

def factorial_recursive(n):
    """
    Calculate factorial using recursive approach.
    Args:
        n: Non-negative integer
    Returns:
        Factorial of n
    """
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers")
    if n == 0 or n == 1:
        return 1
    return n * factorial_recursive(n - 1)

# Test the functions
test_numbers = [0, 1, 5, 10]
for num in test_numbers:
    iterative_result = factorial_iterative(num)
    recursive_result = factorial_recursive(num)
    print(f"Factorial of {num}: {iterative_result} (Iterative), {recursive_result} (Recursive)")
